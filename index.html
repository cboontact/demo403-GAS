3. SCHEMA — กำหนดโครงสร้างข้อมูล

SCHEMA คือ "แผนผังคอลัมน์" ของตาราง — เปรียบได้กับการออกแบบตาราง Database ก่อนสร้าง

4. setupSheet() — ติดตั้งหัวตาราง

ฟังก์ชันนี้รันครั้งเดียวตอนเริ่มโปรเจกต์ เพื่อสร้าง Sheet และหัวตาราง — ถ้ารันซ้ำก็ไม่ลบข้อมูลเดิม

/** รันได้ทุกเมื่อ — สร้างชีต/หัวตารางเท่านั้น ไม่ลบข้อมูลเดิม */
function setupSheet() {

  // 1. ตรวจว่าผู้ใช้เปลี่ยน Sheet ID แล้ว
  validateConfig_();

  // 2. เปิด Spreadsheet ด้วย ID
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

  // 3. หา Sheet ที่ชื่อตรงกับ SHEET_NAME
  let sheet = ss.getSheetByName(SHEET_NAME);

  // 4. ถ้าไม่มี Sheet นั้น → สร้างใหม่
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
  }

  // 4. ดึงชื่อ header ทุกคอลัมน์จาก SCHEMA
  const headers = SCHEMA.map(function(col) { return col.header; });
  //  ผลลัพธ์: ['ID', 'วันที่สมัคร', 'ชื่อ-นามสกุล', ...]

  // 5. เลือก Range แถวที่ 1 ทั้งหมด แล้วใส่ header
  const headerRange = sheet.getRange(1, 1, 1, headers.length);
  //  getRange(แถว, คอลัมน์, จำนวนแถว, จำนวนคอลัมน์)
  headerRange.setValues([headers]);  // ต้องห่อใน Array 2 มิติ [[...]]

  // 6. ตกแต่ง header — สีเขียวเข้ม ตัวหนา ตัวอักษรขาว จัดกลาง
  headerRange
    .setFontWeight('bold')
    .setBackground('#16a34a')
    .setFontColor('#ffffff')
    .setHorizontalAlignment('center');

  // 7. ตรึงแถว header ไว้ด้านบน (Freeze row)
  sheet.setFrozenRows(1);

  // 8. ปรับความกว้างคอลัมน์ให้พอดีข้อมูล
  sheet.autoResizeColumns(1, headers.length);

  // 9. Log ว่ามีข้อมูลเดิมกี่แถว (ตรวจสอบว่าไม่ถูกลบ)
  const dataRows = Math.max(0, sheet.getLastRow() - 1);
  Logger.log('setupSheet: OK — ข้อมูลเดิม ' + dataRows + ' แถว (ไม่ถูกลบ)');
}
Method ที่ใช้อธิบาย
SpreadsheetApp.openById(id)เปิด Spreadsheet จาก ID ใน URL
ss.getSheetByName(name)หา Sheet tab ที่ชื่อตรง — ถ้าไม่เจอ return null
ss.insertSheet(name)สร้าง Sheet tab ใหม่
sheet.getRange(r, c, rows, cols)เลือกพื้นที่เซลล์ (เริ่มนับจาก 1 ไม่ใช่ 0)
range.setValues([[...]])ใส่ข้อมูล — ต้องเป็น Array 2 มิติ
sheet.setFrozenRows(1)ตรึง 1 แถวบนสุด (header จะไม่เลื่อนตามเมื่อ scroll)
sheet.autoResizeColumns(c, n)ปรับความกว้าง n คอลัมน์อัตโนมัติ
sheet.getLastRow()แถวสุดท้ายที่มีข้อมูล — ใช้นับจำนวนแถว
วิธีรัน: ใน Apps Script editor เลือกฟังก์ชัน setupSheet แล้วกดปุ่ม ▶ Run — รันแค่ครั้งเดียว แล้วไปดูผลใน Google Sheet ได้เลย
5. doGet() — จุดเริ่มต้นของ Web App

ฟังก์ชัน doGet() เป็น entry point พิเศษของ GAS — เมื่อมีคนเข้า URL ของ Web App ด้วย HTTP GET, GAS จะเรียกฟังก์ชันนี้โดยอัตโนมัติ

/** แสดงหน้าเว็บฟอร์ม — GAS เรียกอัตโนมัติเมื่อเข้า URL */
function doGet() {
  return HtmlService
    .createHtmlOutputFromFile('index')  // โหลดไฟล์ index.html
    .setTitle('ใบสมัคร ชมรมเขียนเว็บแอป'); // ตั้งชื่อ tab browser
}

6. submitApplication() — บันทึกข้อมูลลง Sheet

ฟังก์ชันนี้ถูกเรียกจาก JavaScript ใน index.html ผ่าน google.script.run — รับ JSON object จากฟอร์ม ตรวจสอบข้อมูล แล้วบันทึกลง Sheet

/** บันทึกข้อมูลจากฟอร์ม — รับ JSON object จาก Frontend */
function submitApplication(data) {

  // ── ① ตรวจสอบว่าได้รับข้อมูลมาจริง และเป็น object ──────────
  if (!data || typeof data !== 'object') {
    return { success: false, error: 'ข้อมูลไม่ถูกต้อง' };
  }

  // ── ② Validate ข้อมูลสำคัญ ──────────────────────────────────
  const name  = trim_(data.name);
  const email = trim_(data.email);
  const phone = trim_(data.phone);

  if (!name)                      return { success: false, error: 'กรุณากรอกชื่อ-นามสกุล' };
  if (name.length > 100)          return { success: false, error: 'ชื่อยาวเกิน 100 ตัวอักษร' };
  if (!email)                     return { success: false, error: 'กรุณากรอกอีเมล' };
  if (!isValidEmail_(email))     return { success: false, error: 'รูปแบบอีเมลไม่ถูกต้อง' };
  if (phone && !/^[0-9+\-\s]{8,20}$/.test(phone)) return { success: false, error: 'รูปแบบเบอร์โทรศัพท์ไม่ถูกต้อง' };
  if (trim_(data.bio).length > 1000) return { success: false, error: 'ข้อความแนะนำตัวยาวเกิน 1,000 ตัวอักษร' };

  // ── ③ เตรียมข้อมูลแต่ละคอลัมน์ตาม SCHEMA ──────────────────
  const sheet = getSheet_();
  const row = SCHEMA.map(function(col) {
    switch (col.key) {

      case 'id':
        // สร้าง UUID อัตโนมัติ เช่น "a3f9b2c1-..."
        return Utilities.getUuid();

      case 'timestamp':
        // บันทึกเวลา ณ ปัจจุบัน รูปแบบ yyyy-MM-dd HH:mm:ss
        return Utilities.formatDate(
          new Date(),
          Session.getScriptTimeZone() || 'Asia/Bangkok',
          'yyyy-MM-dd HH:mm:ss'
        );

      case 'skills':
        // skills เป็น Array เช่น ['html','css','js']
        // เชื่อมเป็น string: "html, css, js"
        return safeText_(Array.isArray(data.skills)
          ? data.skills.join(', ')
          : data.skills);

      default:
        // ตัดช่องว่างและป้องกันค่าที่ Sheet อาจตีความเป็นสูตร
        return safeText_(data[col.key]);
    }
  });
  // row = ['uuid...', '2025-01-15 10:30:00', 'สมชาย', 'som@...', ...]

  // ── ④ เพิ่มแถวใหม่ใน Sheet ──────────────────────────────────
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    sheet.appendRow(row);
  } finally {
    lock.releaseLock();
  }
  // Lock ป้องกันคำขอหลายรายการเขียนแถวพร้อมกัน

  // ── ⑤ ส่งผลกลับ Frontend ──────────────────────────────────
  return { success: true, message: 'บันทึกข้อมูลสำเร็จ' };
}
Method / Utilityอธิบาย
Utilities.getUuid()สร้าง UUID แบบสุ่ม เช่น a3f9b2c1-4b3e-4f2d-... ใช้เป็น ID ไม่ซ้ำ
Utilities.formatDate(date, tz, fmt)แปลง Date เป็น String ตามรูปแบบที่กำหนด
Session.getScriptTimeZone()ดึง timezone ของ Script (ตั้งค่าใน Project Settings)
Array.isArray(val)ตรวจว่าค่าเป็น Array หรือไม่
arr.join(', ')เชื่อม Array เป็น string คั่นด้วย ,
SCHEMA.map(fn)วนลูปทุก column สร้างแถวข้อมูลตามลำดับ SCHEMA
sheet.appendRow(row)เพิ่มแถวใหม่ต่อท้ายข้อมูลที่มีอยู่ใน Sheet
LockService.getScriptLock()จัดคิวการเขียนข้อมูล ป้องกันหลายคำขอแก้ Sheet พร้อมกัน
Flow การส่งข้อมูล: ฟอร์ม → JS เก็บเป็น JSON → google.script.run.submitApplication(json) → GAS ตรวจ → appendRow() → ส่ง {success: true} กลับ → JS แสดง Toast
7. Helper Functions — ฟังก์ชันช่วยเหลือ

ฟังก์ชันที่ชื่อลงท้ายด้วย _ เป็น convention (ข้อตกลง) บอกว่าเป็น private function ใช้ภายในไฟล์เดียวกันเท่านั้น

// ── Helper 1: ตรวจค่าตั้งค่าที่ต้องแก้ก่อนใช้งาน ──────────
function validateConfig_() {
  if (!SPREADSHEET_ID || SPREADSHEET_ID === 'ใส่_SHEET_ID_ของคุณ_ตรงนี้') {
    throw new Error('กรุณาเปลี่ยน SPREADSHEET_ID เป็น Sheet ID ของคุณก่อน');
  }
}

// ── Helper 2: ดึง Sheet — ถ้าไม่มีให้ throw Error ──────────
function getSheet_() {
  validateConfig_();
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    // throw = หยุดทำงานทันที ส่ง Error กลับ Frontend
    throw new Error('ยังไม่ได้ติดตั้งชีต — กรุณารัน setupSheet() ก่อน');
  }
  return sheet;
}

// ── Helper 3: ตัดช่องว่างหัวท้าย ป้องกัน null/undefined ────
function trim_(val) {
  // ถ้า val เป็น null หรือ undefined → return ''
  // ถ้าไม่ใช่ → แปลงเป็น String แล้วตัดช่องว่าง
  return val == null ? '' : String(val).trim();
}
// ตัวอย่าง: trim_('  สมชาย  ') → 'สมชาย'
// ตัวอย่าง: trim_(null)       → ''
// ตัวอย่าง: trim_(undefined)  → ''

// ── Helper 4: ป้องกันข้อความถูกตีความเป็นสูตรใน Sheet ─────
function safeText_(val) {
  const text = trim_(val);
  return /^[=+\-@]/.test(text) ? "'" + text : text;
}

// ── Helper 5: ตรวจรูปแบบอีเมลด้วย Regular Expression ───────
function isValidEmail_(email) {
  // Regex อธิบาย:
  // ^[^\s@]+   = ต้องขึ้นต้นด้วยตัวอักษรที่ไม่ใช่ space หรือ @
  // @          = ต้องมี @
  // [^\s@]+    = ตามด้วยชื่อ domain ไม่มี space หรือ @
  // \.         = ต้องมีจุด .
  // [^\s@]+$   = ลงท้ายด้วย extension เช่น .com .th
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}
// ตัวอย่าง: isValidEmail_('som@gmail.com') → true
// ตัวอย่าง: isValidEmail_('somsak')        → false

getSheet_()—Sheet object หรือ throw Errorเปิด Sheet ทุกครั้งที่ต้องการเขียนข้อมูล
trim_(val)ค่าใดก็ได้String ไม่มีช่องว่างหัวท้ายป้องกัน null crash และช่องว่างในข้อมูล
safeText_(val)ค่าใดก็ได้String ที่ปลอดภัยป้องกันข้อความจากฟอร์มถูกตีความเป็นสูตร

 * Web App ชมรมเขียนเว็บแอป — โรงเรียนจอมทอง
 * บันทึกข้อมูลลง Google Sheet (JSON only)

// ⚠️ ต้องเปลี่ยนเป็น Sheet ID ของคุณก่อนใช้งาน
const SPREADSHEET_ID = 'ใส่_SHEET_ID_ของคุณ_ตรงนี้';
const SHEET_NAME     = 'Applications';

const SCHEMA = [
  { key: 'id',        header: 'ID'           },
  { key: 'timestamp', header: 'วันที่สมัคร'  },
  { key: 'name',      header: 'ชื่อ-นามสกุล' },
  { key: 'email',     header: 'อีเมล'        },
  { key: 'phone',     header: 'เบอร์โทรศัพท์' },
  { key: 'gender',    header: 'เพศ'          },
  { key: 'grade',     header: 'ระดับชั้น'    },
  { key: 'room',      header: 'ห้องเรียน'    },
  { key: 'skills',    header: 'ทักษะ'        },
  { key: 'bio',       header: 'แนะนำตัว'     }
];

/** รันได้ทุกเมื่อ — สร้างชีต/หัวตารางเท่านั้น ไม่ลบข้อมูลเดิม */
function setupSheet() {
  validateConfig_();
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); }
  const headers = SCHEMA.map(function(col) { return col.header; });
  const headerRange = sheet.getRange(1, 1, 1, headers.length);
  headerRange.setValues([headers]);
  headerRange
    .setFontWeight('bold')
    .setBackground('#16a34a')
    .setFontColor('#ffffff')
    .setHorizontalAlignment('center');
  sheet.setFrozenRows(1);
  sheet.autoResizeColumns(1, headers.length);
  const dataRows = Math.max(0, sheet.getLastRow() - 1);
  Logger.log('setupSheet: OK — ข้อมูลเดิม ' + dataRows + ' แถว (ไม่ถูกลบ)');
}

/** แสดงหน้าเว็บฟอร์ม */
function doGet() {
  return HtmlService
    .createHtmlOutputFromFile('index')
    .setTitle('ใบสมัคร ชมรมเขียนเว็บแอป');

/** บันทึกข้อมูลจากฟอร์ม (รับ JSON object) */
function submitApplication(data) {
  if (!data || typeof data !== 'object') {
    return { success: false, error: 'ข้อมูลไม่ถูกต้อง' };
  }
  const name  = trim_(data.name);
  const email = trim_(data.email);
  const phone = trim_(data.phone);
  if (!name)               return { success: false, error: 'กรุณากรอกชื่อ-นามสกุล' };
  if (name.length > 100)   return { success: false, error: 'ชื่อยาวเกิน 100 ตัวอักษร' };
  if (!email)              return { success: false, error: 'กรุณากรอกอีเมล' };
  if (!isValidEmail_(email)) return { success: false, error: 'รูปแบบอีเมลไม่ถูกต้อง' };
  if (phone && !/^[0-9+\-\s]{8,20}$/.test(phone)) return { success: false, error: 'รูปแบบเบอร์โทรศัพท์ไม่ถูกต้อง' };
  if (trim_(data.bio).length > 1000) return { success: false, error: 'ข้อความแนะนำตัวยาวเกิน 1,000 ตัวอักษร' };
  const sheet = getSheet_();
  const row = SCHEMA.map(function(col) {
    switch (col.key) {
      case 'id':        return Utilities.getUuid();
      case 'timestamp': return Utilities.formatDate(new Date(), Session.getScriptTimeZone() || 'Asia/Bangkok', 'yyyy-MM-dd HH:mm:ss');
      case 'skills':    return safeText_(Array.isArray(data.skills) ? data.skills.join(', ') : data.skills);
      default:         return safeText_(data[col.key]);
    }
  });
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    sheet.appendRow(row);
  } finally {
    lock.releaseLock();
  }
  return { success: true, message: 'บันทึกข้อมูลสำเร็จ' };
}

function validateConfig_() {
  if (!SPREADSHEET_ID || SPREADSHEET_ID === 'ใส่_SHEET_ID_ของคุณ_ตรงนี้') {
    throw new Error('กรุณาเปลี่ยน SPREADSHEET_ID เป็น Sheet ID ของคุณก่อน');
  }
}

function getSheet_() {
  validateConfig_();
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) { throw new Error('ยังไม่ได้ติดตั้งชีต — กรุณารัน setupSheet() ก่อน'); }
  return sheet;
}

function trim_(val) {
  return val == null ? '' : String(val).trim();
}

function safeText_(val) {
  const text = trim_(val);
  return /^[=+\-@]/.test(text) ? "'" + text : text;
}

function isValidEmail_(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

// ── ② fillSelect — เติม option ใน <select> ────────────────
function fillSelect(el, placeholder, items) {
  // สร้าง option แรก (placeholder)
  el.innerHTML = '<option value="">' + placeholder + '</option>';
  // วนลูปสร้าง option ตาม items array
  items.forEach(function(item) {
    const opt = document.createElement('option');
    opt.value       = item.value;
    opt.textContent = item.label;
    el.appendChild(opt);
  });
}

// ── ④ showToast — แสดงกล่องแจ้งเตือน ──────────────────────
function showToast(type, message) {
  const isSuccess = type === 'success';
  const toast = document.createElement('div');
  toast.className = 'toast ' + type;
  toast.innerHTML = /* ... สร้าง HTML ของ toast ... */
    '<div class="toast-icon"><i class="fa-solid fa-' +
    (isSuccess ? 'circle-check' : 'circle-xmark') + '"></i></div>' +
    '<div class="toast-body"><div class="toast-title">' +
    (isSuccess ? 'สำเร็จ' : 'ล้มเหลว') +
    '</div><div>' + message + '</div></div>' +
    '<button class="toast-close"><i class="fa-solid fa-xmark"></i></button>';

  const close = function() {
    toast.classList.add('hide');               // เล่น animation slide out
    setTimeout(function() { toast.remove(); }, 300); // ลบออกหลัง animation
  };
  toast.querySelector('.toast-close').addEventListener('click', close);
  toastContainer.appendChild(toast);
  setTimeout(close, 4000); // ปิดอัตโนมัติหลัง 4 วินาที
}

// ── ⑤ collectFormJson — รวบรวมข้อมูลจากฟอร์มเป็น JSON ────
function collectFormJson() {
  // checkbox ที่ติ๊ก → เก็บเป็น Array
  const skills = Array.from(
    document.querySelectorAll('input[name="skill"]:checked')
  ).map(function(cb) { return cb.value; });
  // ถ้าติ๊ก html,css,js → skills = ['html', 'css', 'js']

  // radio → เลือกตัวที่ checked
  const genderEl = document.querySelector('input[name="gender"]:checked');

  return {
    name:   document.getElementById('name').value.trim(),
    email:  document.getElementById('email').value.trim(),
    phone:  document.getElementById('phone').value.trim(),
    gender: genderEl ? genderEl.value : '',  // ถ้าไม่เลือก → ''
    grade:  document.getElementById('grade').value,
    room:   document.getElementById('room').value,
    skills: skills,
    bio:    document.getElementById('bio').value.trim()
  };
}

// ── ⑥ submit — ส่งข้อมูลไป GAS ─────────────────────────────
form.addEventListener('submit', function(e) {
  e.preventDefault();  // ป้องกัน browser refresh หน้า
  setButtonLoading(true);

  const payload = collectFormJson();

  // google.script.run — Bridge ระหว่าง HTML และ GAS
  google.script.run
    .withSuccessHandler(function(res) {   // GAS return ปกติ
      setButtonLoading(false);
      if (res.success) {
        showToast('success', res.message);
        form.reset();               // ล้างฟอร์ม
      } else {
        showToast('error', res.error || 'เกิดข้อผิดพลาด');
      }
    })
    .withFailureHandler(function(err) {  // GAS throw Error
      setButtonLoading(false);
      showToast('error', err.message || 'ไม่สามารถเชื่อมต่อเซิร์ฟเวอร์ได้');
    })
    .submitApplication(payload);  // เรียกฟังก์ชัน GAS
});

9. วิธี Deploy — เผยแพร่เป็น Web App
1
เปิด Google Sheet → คัดลอก ID จาก URL
https://docs.google.com/spreadsheets/d/ID_ตรงนี้/edit
→ วางใน SPREADSHEET_ID ใน code.gs
2
สร้าง Google Apps Script Project
ใน Google Drive → New → More → Google Apps Script
วางโค้ด code.gs → บันทึก (Ctrl+S)
3
สร้างไฟล์ HTML
ใน Script Editor → กด + → HTML → ตั้งชื่อ index (ห้ามมี .html)
วางโค้ด index.html → บันทึก
4
รัน setupSheet()
เลือกฟังก์ชัน setupSheet → กด ▶ Run → เลือกบัญชี Google
ตรวจชื่อ Project → อนุญาตสิทธิ์เข้าถึง Spreadsheet → กลับมาดู Execution log
→ ไปดู Google Sheet และตรวจว่ามีแท็บ Applications พร้อม header แถวแรก
5
Deploy เป็น Web App
Deploy → New deployment → Web app
Execute as: Me | Who has access: Anyone
→ Copy URL ที่ได้ → เปิดในบราวเซอร์ได้เลย
เมื่อแก้ code.gs หรือ index.html: URL สำหรับใช้งานจริงที่ลงท้ายด้วย /exec จะยังใช้ Version เดิม ต้องไปที่ Deploy → Manage deployments → Edit → เลือก New version → Deploy จึงจะเห็นโค้ดใหม่โดยใช้ URL เดิม
URL ทดสอบกับ URL ใช้งานจริง: Test deployment ที่ลงท้ายด้วย /dev ใช้โค้ดที่บันทึกล่าสุดและเปิดได้เฉพาะผู้มีสิทธิ์แก้ Script ส่วน URL /exec ใช้ Version ที่ Deploy แล้วและใช้แจกให้ผู้ใช้จริง

Checklist ก่อนส่ง URL ให้นักเรียนทดสอบ

  • เปลี่ยน SPREADSHEET_ID เป็น ID ของ Google Sheets ของคุณแล้ว
  • ตรวจค่า SHEET_NAME และสร้างไฟล์ HTML ชื่อ index แล้ว
  • ตั้ง Time zone ใน Project Settings เป็น Asia/Bangkok
  • รัน setupSheet() สำเร็จและเห็น header ใน Sheet
  • ทดสอบส่งข้อมูลอย่างน้อย 1 รายการผ่าน /dev
  • Deploy Version ล่าสุด แล้วทดสอบซ้ำผ่าน URL /exec
  • ตรวจว่าข้อมูลถูกบันทึกครบทุกคอลัมน์ใน Google Sheets
10. ข้อผิดพลาดที่พบบ่อยและวิธีตรวจสอบ
อาการ / ข้อความสาเหตุที่พบบ่อยวิธีแก้
กรุณาเปลี่ยน SPREADSHEET_ID...ยังใช้ข้อความตัวอย่างแทน Sheet ID จริงคัดลอกค่าระหว่าง /d/ และ /edit จาก URL ของ Google Sheets
Unexpected error while getting openByIdSheet ID ผิด หรือบัญชีเจ้าของ Script ไม่มีสิทธิ์เปิดไฟล์ตรวจ ID และเปิด Google Sheets ด้วยบัญชีเดียวกับที่ Deploy
Cannot find file indexไม่มีไฟล์ HTML ชื่อ index หรือสะกดไม่ตรงสร้างไฟล์ HTML ชื่อ index โดยไม่ต้องพิมพ์นามสกุล
หน้าเว็บยังเป็นโค้ดเก่าเปิด URL /exec ที่ยังชี้ไป Version เดิมอัปเดต Deployment ให้ใช้ New version แล้ว Refresh หน้า
ส่งข้อมูลแล้วขึ้น Permission errorยังไม่ได้อนุญาตสิทธิ์ หรือ Execute as ตั้งค่าไม่เหมาะสมรัน setupSheet() เพื่ออนุญาตสิทธิ์ และตรวจว่า Execute as เป็น Me
บันทึกสำเร็จแต่หาแถวไม่พบSheet ID หรือ SHEET_NAME ชี้ไปคนละไฟล์/คนละแท็บตรวจค่าคงที่ทั้งสองตัวและดูแท็บ Applications
จุดตรวจสอบหลัก: ใน Apps Script เปิดเมนู Executions เพื่อดูว่าฟังก์ชันใดทำงาน เวลาใด สำเร็จหรือผิดพลาด และอ่านข้อความ Error ฉบับเต็ม
11. ขอบเขตการใช้งานและข้อจำกัด

ตัวอย่างนี้เหมาะสำหรับการเรียน การเก็บแบบฟอร์ม และระบบภายในที่มีผู้ใช้ไม่มาก Google Sheets ช่วยให้เปิดดูและแก้ข้อมูลได้ง่าย แต่ไม่ใช่ฐานข้อมูลสำหรับงานที่ต้องรองรับปริมาณข้อมูลหรือคำขอจำนวนมากมาก

  • Google Apps Script และบริการ Google มีโควตาการใช้งาน ซึ่งแตกต่างตามประเภทบัญชีและอาจมีการเปลี่ยนแปลง
  • ไม่ควรเก็บรหัสผ่าน เลขบัตรประชาชน หรือข้อมูลอ่อนไหวโดยไม่มีมาตรการรักษาความปลอดภัยและสิทธิ์เข้าถึงที่เหมาะสม
  • การตั้ง Who has access เป็น Anyone หมายถึงทุกคนที่มี URL สามารถเปิดฟอร์มได้ ควรแจก URL เฉพาะกลุ่มที่ต้องการ
  • หากระบบโตขึ้น ควรพิจารณาฐานข้อมูลจริง ระบบยืนยันตัวตน และการสำรองข้อมูล
